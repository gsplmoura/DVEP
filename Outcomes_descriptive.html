baseline <- data_model %>%
    filter(visit == 1)

summ_cat <- baseline %>%
  summarize_categorical(use_labels = TRUE, group_col = "allocation_group") %>%
  filter(Level == 1) %>%
  select(Variable, Freq, Percent, allocation_group) %>%
  pivot_wider(
    names_from = allocation_group, 
    values_from = c(Freq, Percent)
    )

summ_cat <- summ_cat %>% 
    select(
        Variable, 
        `Grupo Placebo (N)` = `Freq_Grupo A`, 
        `% (A)` = `Percent_Grupo A`,
